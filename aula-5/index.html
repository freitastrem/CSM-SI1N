            import numpy as np
import pandas as pd

# Exemplo de dados
data = {'valores': [10, 12, 12, 13, 12, 14, 14, 15, 100]}  # Incluindo um outlier
df = pd.DataFrame(data)

# Cálculo do IQR
Q1 = df['valores'].quantile(0.25)
Q3 = df['valores'].quantile(0.75)
IQR = Q3 - Q1

# Definindo limites para identificar outliers
limite_inferior = Q1 - 1.5 * IQR
limite_superior = Q3 + 1.5 * IQR

# Identificando outliers
df['outlier'] = (df['valores'] < limite_inferior) | (df['valores'] > limite_superior)

# Exibindo resultados
print(df)
